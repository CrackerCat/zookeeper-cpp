Redirect the provided fd to read from this pipe instead of what it was originally reading from.This redirects standard input to be controlled by a pipe. See subprocess for the use case.

pipe p;
p.subsume_read(STDIN_FILENO);
// Now p.write(...) calls will be picked up by reads to standard input
#pragma once
#include <zk/config.hpp>
#include <zk/optional.hpp>
#include <stdexcept>
#include <string>
namespace zk::server::detail
{
enum class on_exec
{
close,
keep_open,
};
class pipe_closed :
public std::runtime_error
{
public:
pipe_closed();
virtual ~pipe_closed() noexcept;
};
class pipe final
{
public:
using handle = int;
public:
explicit pipe(on_exec exec = on_exec::close);
pipe(const pipe&) = delete;
pipe& operator=(const pipe&) = delete;
~pipe() noexcept;
void close();
void close_read();
void close_write();
std::string read(optional<std::size_t> max = nullopt);
void write(const std::string& contents);
void subsume_read(handle fd, on_exec exec = on_exec::close);
void subsume_write(handle fd, on_exec exec = on_exec::close);
handle native_read_handle();
handle native_write_handle();
private:
handle _read_fd;
handle _write_fd;
};
}
